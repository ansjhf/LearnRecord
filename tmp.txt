import com.alibaba.excel.EasyExcel;
import com.alibaba.excel.context.AnalysisContext;
import com.alibaba.excel.event.AnalysisEventListener;

import java.io.File;
import java.util.ArrayList;
import java.util.List;

public class ExcelReader {

    public static List<List<String>> readExcelToList(String filePath) {
        List<List<String>> dataList = new ArrayList<>();

        EasyExcel.read(filePath, new AnalysisEventListener<List<String>>() {
            @Override
            public void invoke(List<String> data, AnalysisContext context) {
                dataList.add(data);
            }

            @Override
            public void doAfterAllAnalysed(AnalysisContext context) {
                // 所有数据解析完成
            }
        }).sheet().doRead();

        return dataList;
    }

    public static void main(String[] args) {
        String filePath = "test.xlsx";
        List<List<String>> result = readExcelToList(filePath);

        for (List<String> row : result) {
            System.out.println(row);
        }
    }
}
